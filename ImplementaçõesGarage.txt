  "DATABASE_URL": "...",

  "RESEND_API_KEY": "..."
}
1.3 Loader centralizado
# app/infrastructure/config/secret_loader.py

import os
import json
import boto3

def load_secrets():
    if os.getenv("ENV") == "local":
        return

    client = boto3.client("secretsmanager")
    secret = client.get_secret_value(SecretId="garage/prod")
    data = json.loads(secret["SecretString"])

    for k, v in data.items():
        os.environ[k] = v
1.4 Bootstrap no startup

3. 🧠 IDEMPOTÊNCIA DISTRIBUÍDA (CRITICAL FOR SCALE)
Objetivo

Garantir exactly-once semantics

Implementação
3.1 Tabela com constraint
CREATE TABLE idempotency_keys (
    key TEXT PRIMARY KEY,
    response JSONB,
    created_at TIMESTAMP DEFAULT now()
);
3.2 Middleware (DB-backed only)
def process_request(key, handler):
    existing = db.get(key)

    if existing:
        return existing.response

    response = handler()

    db.insert(key, response)
    return response
3.3 Versão distribuída (Redis)

3.4 Regra obrigatória
